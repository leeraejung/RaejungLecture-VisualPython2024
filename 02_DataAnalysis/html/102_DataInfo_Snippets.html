def as_int(x):
    """
    usage: df['col'] = df['col'].apply(as_int)
    """
    if not isinstance(x, str):
        return 0
    else:
        try:
            result = int(x)
            return result
        except ValueError:
            return 0.0

def as_int(x):
    """
    usage: df['col'] = df['col'].apply(as_int)
    """
    if not isinstance(x, str):
        return 0
    else:
        try:
            result = int(x)
            return result
        except ValueError:
            return 0.0
